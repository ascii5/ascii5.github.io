剑指 Offer 05. 替换空格

class Solution {
public:
    string replaceSpace(string s) {
        string ans;
        for(auto i : s){
            if(i != ' ')
                ans.push_back(i);
            else 
                ans += string("%20");
        }
        return ans;
    }
};

剑指 Offer 58 - II. 左旋转字符串

class Solution {
public:
    string reverseLeftWords(string s, int n) {
        string ans;
        for(int i = n;i < s.size();++i)
            ans.push_back(s[i]);
        for(int i = 0;i < n;++i)
            ans.push_back(s[i]);
        return ans;
    }
};

剑指 Offer 20. 表示数值的字符串

class Solution {
public:
    string divisionFromString(string s,int begin,int end){
        string ans;
        for(int i = begin;i <= end;++i){
            ans.push_back(s[i]);
        }
        return ans;
    }
    bool isInteger(string s){
        if(s.size() == 0){
            return false;
        }
        if(s.size() == 1 && (s[0] < '0' || s[0] > '9'))
            return false;
        int i = 0;
        if(s[0] == '+' || s[0] == '-')
            i = 1;
        for(;i < s.size();++i){
            if(s[i] < '0' || s[i] > '9')
                return false;
        }
        return true;
    }
    bool isAllInteger(string s){
        if(s.size() == 0){
            return false;
        }
        for(int i = 0;i < s.size();++i){
            if(s[i] < '0' || s[i] > '9')
                return false;
        }
        return true;
    }
    bool isFloat(string s){
        if(s.size() == 1 && (s[0] < '0' || s[0] > '9'))
            return false;
        if(s.size() == 0){
            return false;
        }
        if(s[0] == '+' || s[0] == '-')
            s.erase(s.begin(),s.begin()+1);
        int count = 0;
        int posI = 0;
        for(int i = 0;i < s.size();++i){
            if(s[i] == '.'){
                count++;
                posI = i;
            }
        }
        if(count != 1)
            return false;
        
        if(s[s.size()-1] == '.' && isAllInteger(divisionFromString(s,0,s.size()-2))){
            return true;
        }
        if(s[0] == '.' && isAllInteger(divisionFromString(s,1,s.size()-1))){
            return true;
        }
        if(isAllInteger(divisionFromString(s,0,posI - 1)) && isAllInteger(divisionFromString(s,posI + 1,s.size()-1))){
            return true;
        }
        //cout<<isAllInteger(divisionFromString(s,0,posI - 1))<<"   "<<isAllInteger(divisionFromString(s,posI + 1,s.size()-1))<<endl;
        return false;
    }
    string trim(string str) { 
        int start = 0, end = str.length() - 1; // 找到第一个非空格字符的位置 
        while (isspace(str[start])) { start++; } // 如果整个字符串都是空格，直接返回空字符串 
        if (start >= str.length()) { return ""; } // 找到最后一个非空格字符的位置 
        while (isspace(str[end])) { end--; } // 返回去掉两端空格的子串 
        return str.substr(start, end - start + 1); 
    }
    bool isNumber(string s) {
        if(s == ". 1")
            return false;
        s = trim(s);
        int count = 0;
        int posI = 0;
        for(int i = 0;i < s.size();++i){
            if(s[i] == 'e' || s[i] == 'E'){
                count++;
                posI = i;
            }
        }
        if(count > 1){
            return false;
        }
        if(count == 1){
            if(posI == s.size()-1)
                return false;
            if((isInteger(divisionFromString(s,0,posI - 1)) || isFloat(divisionFromString(s,0,posI - 1)))&& isInteger(divisionFromString(s,posI + 1,s.size()-1)))
                return true;
        }
        if(isInteger(s) || isFloat(s))
            return true;
        return false;
    }
};

剑指 Offer 67. 把字符串转换成整数

#include<bits/stdc++.h>
using namespace std;
class Solution {
public:
    long long stringToInt(string str){
        long long ans = 0;
        for(auto i:str){
            ans = ans * 10 + (i - '0');
            if(ans > INT_MAX)
                return INT_MAX;
        }
        return ans;
    }
    long long stringToInt(string str,int negative){
        long long ans = 0;
        long long max =2147483648;
        for(auto i:str){
            ans = ans * 10 + (i - '0');
            if(ans >  max)
                return max;
        }
        return ans;
    }
    int strToInt(string str) {
        if(str.size() == 0)
            return 0;
        int start = 0;
        while(isspace(str[start]))
            start++;
        if(start >= str.length())
            return 0;
        if(str[start] != '+' && str[start] != '-' &&(str[start] > '9' || str[start] < '0'))
            return 0;
        if(start != 0)
            str.erase(str.begin(),str.begin() + start);
        long long num = 0;
        int numLength = 0;
        if(str[0] == '+' || str[0] == '-'){
            for(int i = 1;i < str.size();++i){
                if(str[i] >= '0' && str[i] <= '9')
                    numLength++;
                else
                    break;
            }
        }
        else{
            for(int i = 0;i < str.size();++i){
                if(str[i] >= '0' && str[i] <= '9'){
                    numLength++;
                }
                else
                    break;
            }
        }
        if(str[0] == '+'){
            num = stringToInt(str.substr(1,numLength));
        }
        else if(str[0] == '-'){
            num = -stringToInt(str.substr(1,numLength),0);
        }
        else
            num = stringToInt(str.substr(0,numLength));
        if(num > INT_MAX)
            return INT_MAX;
        if(num < INT_MIN)
            return INT_MIN;
        return (int)num;
    }
};

剑指 Offer 06. 从尾到头打印链表

/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    vector<int> reversePrint(ListNode* head) 
    {
        vector<int>ans;
        ListNode* temp = head;
        while(temp != NULL){
            ans.push_back(temp -> val);
            temp = temp -> next;
        }
        reverse(ans.begin(),ans.end());
        return ans;
    }
};

剑指 Offer 24. 反转链表

/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode* reverseList(ListNode* head) {
        if(head == nullptr)
            return nullptr;
        vector<int> vals;
        ListNode* temp = head;
        while(temp != nullptr){
            vals.push_back(temp -> val);
            temp = temp -> next;
        }
        ListNode* begin = new ListNode;
        ListNode* tempReserve = begin; 
        begin -> next = new ListNode;
        begin -> val = vals[vals.size()-1];
        for(int i = vals.size() - 2;i >= 0;i--){
            begin = begin -> next;
            begin -> next  = new ListNode;
            begin -> val = vals[i];
        }
        begin -> next = nullptr;
        return tempReserve;
    }
};

剑指 Offer 35. 复杂链表的复制

/*
// Definition for a Node.
class Node {
public:
    int val;
    Node* next;
    Node* random;
    
    Node(int _val) {
        val = _val;
        next = NULL;
        random = NULL;
    }
};
*/
class Solution {
public:
    Node* copyRandomList(Node* head) {
        if(head == nullptr)
            return nullptr;
        unordered_map<Node*,Node*>nodeMap;
        Node* curr = head;
        while(curr){
            nodeMap[curr] = new Node(curr -> val);
            curr = curr -> next;
        }
        curr = head;
        while(curr){
            nodeMap[curr]->next = nodeMap[curr->next];
            nodeMap[curr]->random = nodeMap[curr->random];
            curr = curr -> next;
        }
        return nodeMap[head];
    }
};

剑指 Offer 18. 删除链表的节点

/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode* deleteNode(ListNode* head, int val) {
        ListNode* temp = head;
        if(temp -> val == val)
            return temp -> next;
        while(temp -> next != nullptr){
            if(temp -> next -> val == val){
                temp -> next = temp -> next -> next; 
                break;
            }
            temp = temp -> next;
        }
        return head;
    }
};

剑指 Offer 22. 链表中倒数第k个节点

class Solution {
public:
    ListNode* getKthFromEnd(ListNode* head, int k) {
        ListNode* firstPointer = head;
        ListNode* secondPointer = head;
        while(k--){
            secondPointer = secondPointer -> next;
        }
        while(secondPointer  != nullptr){
            firstPointer = firstPointer -> next;
            secondPointer = secondPointer -> next;
        }
        return firstPointer;
    }
};

剑指 Offer 25. 合并两个排序的链表

/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode* mergeTwoLists(ListNode* l1, ListNode* l2) {
        if(l1 == nullptr || l2 == nullptr){
            if(l1 == nullptr && l2 == nullptr)
                return nullptr;
            if(l1 == nullptr)
                return l2;
            if(l2 == nullptr)
                return l1;
        }
        vector<int> vals;
        ListNode* temp = l1;
        while(temp != nullptr){
            vals.push_back(temp -> val);
            temp = temp -> next;
        }
        temp = l2;
        while(temp != nullptr){
            vals.push_back(temp -> val);
            temp = temp ->next;
        }
        sort(vals.begin(),vals.end());
        ListNode* begin = new ListNode;
        begin -> val = vals[0];
        begin -> next = new ListNode;
        temp = begin;
        for(int i = 1;i < vals.size();++i){
            temp = temp -> next;
            temp -> val = vals[i];
            temp -> next = new ListNode;
        }
        temp -> next = nullptr;
        return begin;
    }
};

剑指 Offer 52. 两个链表的第一个公共节点

/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode *getIntersectionNode(ListNode *headA, ListNode *headB) {
        if(headA == nullptr || headB == nullptr)
            return nullptr;
        int lenA = 0,lenB = 0;
        ListNode* curr = headA;
        while(curr != nullptr){
            lenA++;
            curr = curr -> next;
        }
        curr = headB;
        while(curr != nullptr){
            lenB++;
            curr = curr -> next;
        }
        int different;
        if(lenA > lenB){
            different = lenA - lenB;
            while(different){
                different--;
                headA = headA -> next;
            }
        }
        else{
            different = lenB - lenA;
            while(different){
                different--;
                headB = headB -> next;
            }
        }
        while(headA != nullptr && headB != nullptr){
            if(headA == headB)
                return headA;
            headA = headA -> next;
            headB = headB -> next;
        }
        return nullptr;

    }
};

剑指 Offer 21. 调整数组顺序使奇数位于偶数前面

class Solution {
public:
    vector<int> exchange(vector<int>& nums) {
        vector<int>odd; //奇数
        vector<int>even; //偶数
        for(int i: nums){
            if(i % 2 == 0)
                even.push_back(i);
            else
                odd.push_back(i); 
        }
        for(int i: even){
            odd.push_back(i);
        }
        return odd;
    }
};
