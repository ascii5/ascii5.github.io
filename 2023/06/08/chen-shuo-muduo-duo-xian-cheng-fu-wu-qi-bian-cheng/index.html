blockingqueue

#include<bits/stdc++.h>
#include<mutex>
#include<condition_variable>
#include<thread>
using namespace std;
class blockingQueue{
public:
    int dequeue(){
        unique_lock<mutex>lock(mutexForque);
        while(myQueue.empty()){
            cond.wait(lock,[this]{ 
                return !myQueue.empty();});
        }
        int top = myQueue.front();
        myQueue.pop();
        return top;
    }
    void enqueue(int num){
        unique_lock<mutex>lock(mutexForque);
        myQueue.push(num);
        cond.notify_one();
    }
private:
    mutex mutexForque;
    condition_variable cond;
    queue<int> myQueue;
};
int main(){
    blockingQueue myqueue;
    thread thread1(blockingQueue::dequeue,&myqueue);
    thread thread2(blockingQueue::dequeue,&myqueue);
    myqueue.enqueue(1);
    myqueue.enqueue(2);
    thread1.join();
    thread2.join();
    cout<<"all threads are completed !"<<endl;    
    return 0;
}
